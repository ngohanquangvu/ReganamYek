= requests.get(f"{cls.BASE_URL}/key_manager", headers=cls.headers) return response.json() @classmethod def add_data(cls, data): response = requests.post(f"{cls.BASE_URL}/admin_edit", json=data, headers=cls.headers) print(response.json()) return response.json() @classmethod def delete_data(cls, key): response = requests.delete(f"{cls.BASE_URL}/admin_edit?key={key}", headers=cls.headers) return response.json() class DataManagementApp: def __init__(self, root): self.root = root self.root.title("Data Management App") self.tree = ttk.Treeview(root, columns=("Status", "User", "Key", "Remain Days", "Refile Date")) self.tree.heading("#1", text="Status") self.tree.heading("#2", text="User") self.tree.heading("#3", text="Key") self.tree.heading("#4", text="Remain Days") self.tree.heading("#5", text="Refile Date") self.tree.pack(padx=10, pady=10) add_button = tk.Button(root, text="Add Data", command=self.add_data_dialog) add_button.pack(pady=5) delete_button = tk.Button(root, text="Delete Selected", command=self.delete_data) delete_button.pack(pady=5) refile_button = tk.Button(root, text="Refile", command=self.refile_data) refile_button.pack(pady=5) self.load_data() def load_data(self): data = APIClient.get_data() for item in data: remain_date_str = item["remain_days"] remain_date = datetime.strptime(remain_date_str, "%d-%m-%Y") remain_days = (remain_date - datetime.now()).days status = "Expired" if remain_days < 0 else "Working" status_color = "red" if remain_days < 0 else "green" self.tree.insert("", "end", values=(status, item["user"], item["key"], remain_date_str, item["refile_date"]), tags=(status,)) self.tree.tag_configure(status, foreground=status_color) def add_data_dialog(self): add_window = tk.Toplevel(self.root) add_window.title("Add Data") user_label = tk.Label(add_window, text="User:") user_label.grid(row=0, column=0, padx=5, pady=5) user_entry = tk.Entry(add_window) user_entry.grid(row=0, column=1, padx=5, pady=5) key_label = tk.Label(add_window, text="Key:") key_label.grid(row=1, column=0, padx=5, pady=5) key_entry = tk.Entry(add_window) key_entry.grid(row=1, column=1, padx=5, pady=5) remain_days_label = tk.Label(add_window, text="Remain Days:") remain_days_label.grid(row=2, column=0, padx=5, pady=5) remain_days_entry = tk.Entry(add_window) remain_days_entry.grid(row=2, column=1, padx=5, pady=5) refile_date_label = tk.Label(add_window, text="Refile Date (in days):") refile_date_label.grid(row=3, column=0, padx=5, pady=5) refile_date_entry = tk.Entry(add_window) refile_date_entry.grid(row=3, column=1, padx=5, pady=5) add_button = tk.Button(add_window, text="Add", command=lambda: self.add_data( user_entry.get(), key_entry.get(), remain_days_entry.get(), refile_date_entry.get(), add_window)) add_button.grid(row=4, column=0, columnspan=2, pady=10) def add_data(self, user, key, remain_days, refile_date, add_window): try: remain_days = int(remain_days) refile_days = int(refile_date) refile_date = datetime.now() + timedelta(days=refile_days) refile_date_str = refile_date.strftime("%d-%m-%Y") remain_date = datetime.now() + timedelta(days=remain_days) remain_date_str = remain_date.strftime("%d-%m-%Y") new_data = {"user": user, "key": key, "remain_days": str(remain_date_str), "refile_date": str(refile_date_str)} response = APIClient.add_data(new_data) if response.get("message") == "Data added successfully": status = "Expired" if remain_days < 0 else "Working" status_color = "red" if remain_days < 0 else "green" self.tree.insert("", "end", values=(status, user, key, remain_date_str, refile_date_str), tags=(status,)) self.tree.tag_configure(status, foreground=status_color) add_window.destroy() else: print("Failed to add data:", response.get("message", "Unknown error")) except Exception as e: print(str(e)) print("Invalid input for Remain Days or Refile Date") def delete_data(self): selected_item = self.tree.selection() if not selected_item: print("No item selected for deletion.") return key_to_delete = self.tree.item(selected_item, "values")[2] response = APIClient.delete_data(key_to_delete) if response.get("message") == "Key deleted successfully": self.tree.delete(selected_item) else: print("Failed to delete data:", response.get("message", "Unknown error")) def refile_data(self): selected_item = self.tree.selection() if not selected_item: print("No item selected for refile.") return key_to_refile = self.tree.item(selected_item, "values")[2] # Mở hộp thoại chọn tệp tin file_path = filedialog.askopenfilename(title="Select a file to refile") if file_path: # Thư mục đích là tên của key destination_folder = f"./{key_to_refile}" # Tạo thư mục nếu nó chưa tồn tại if not os.path.exists(destination_folder): os.makedirs(destination_folder) try: # Di chuyển tệp tin đến thư mục đích shutil.move(file_path, os.path.join(destination_folder, os.path.basename(file_path))) print("Refile successful.") except Exception as e: print(f"Refile failed: {str(e)}") if __name__ == "__main__": root = tk.Tk() app = DataManagementApp(root) root.mainloop()
